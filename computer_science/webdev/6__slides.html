style="color: lightseagreen; font-size: 16pt; font-family: fantasy;"

Weiter mit h1:
style="color: darkgreen; font-family: fantasy;"

Weiter mit img:
style="border:3px solid grey; border-radius:90px 0px 90px 0px; box-shadow: 2px 2px 7px 1px #CE5937;"

Weiter mit img:
style="float: left" & auch widht auf 20% setzen

Weiter mit h1 und p:
style="margin-left: 25%; margin-right: 20%"

style="text-align: justify;"
